SET(CMAKE_CXX_STANDARD 11)
SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O3")

# 增加PCL库的依赖

LINK_LIBRARIES( ${PCL_LIBRARY_DIRS} )

INCLUDE_DIRECTORIES( ${OCTOMAP_INCLUDE_DIRS} )
message(${OCTOMAP_LIBRARIES})

)

${DIR_SRCS}
)

    ${PCL_LIBRARIES}
    ${OCTOMAP_LIBRARIES})
